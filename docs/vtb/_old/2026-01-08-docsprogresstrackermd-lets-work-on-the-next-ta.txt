  What is CMBlockBuffer?

  Think of it as: A CoreMedia container that wraps raw bytes and says "here's a block of compressed data"

  The decoder doesn't just want raw pointers—it wants a structured object that:
  - Knows where the bytes are
  - Knows how many bytes there are
  - Handles memory management properly
  - Provides consistent access patterns

  CMBlockBufferCreateWithMemoryBlock() creates one of these containers.

  ---
  The Parameters (Conceptually)

  The Apple API signature looks roughly like:

  CMBlockBufferCreateWithMemoryBlock(
      CFAllocatorRef structureAllocator,    // Allocator for the CMBlockBuffer object itself
      void *memoryBlock,                    // ← Your frame bytes
      size_t blockLength,                   // ← Your frame size
      CFAllocatorRef blockAllocator,        // Who owns the memory?
      const CMBlockBufferCustomBlockSource *customBlockSource,  // (usually NULL)
      size_t offsetToData,                  // Offset within the block where data starts
      size_t dataLength,                    // How many bytes of actual frame data
      CMBlockBufferFlags flags,             // Options (usually 0)
      CMBlockBufferRef *blockBufferOut      // ← Output: the created object
  )

  The core idea: You're saying "I have this block of memory with N bytes. The actual frame data is M bytes starting at offset O."

      timescale: u32,      // Denominator
      flags: u32,          // Metadata flags

  };

  fn CMTimeMake(value: i64, timescale: u32) CMTime {
      return .{ .value = value, .timescale = timescale, .flags = 1, .epoch = 0 };

  For PTS of frame N, it would be: CMTimeMake(N × frame_rate.den, frame_rate.num)

  fn createSampleTimingInfo(
      frame_index: usize,
      source_media: media.SourceMedia,
  ) vtb.CMSampleTimingInfo {
      // PTS = frame_index × frame_duration / timescale
      // For frame 0: PTS = 0
      const pts_value: i64 = @as(i64, @intCast(frame_index)) * @as(i64, @intCast(source_media.frame_rate.den));
      const pts = vtb.CMTimeMake(pts_value, source_media.frame_rate.num);

      // Duration = frame_duration / timescale
      const duration = vtb.CMTimeMake(@as(i64, @intCast(source_media.frame_rate.den)), source_media.frame_rate.num);

      // DTS (decode timestamp) is usually same as PTS for intraframe codecs like ProRes
      const dts = pts;

      return .{
          .duration = duration,
          .presentationTimeStamp = pts,
          .decodeTimeStamp = dts,
      };
  }

  fn createSampleBuffer(
      block_buffer: vtb.CMBlockBufferRef,
      timing_info: vtb.CMSampleTimingInfo,
      format_desc: vtb.CMVideoFormatDescriptionRef,
  ) !vtb.CMSampleBufferRef {
      var sample_buffer: vtb.CMSampleBufferRef = null;

      const status = vtb.CMSampleBufferCreateReady(
          vtb.kCFAllocatorDefault,        // allocator
          block_buffer,                   // dataBuffer (our block buffer)
          format_desc,                    // formatDescription
          1,                              // numSamples (1 frame)
          1,                              // numSampleTimingEntries
          &timing_info,                   // sampleTimingArray
          0,                              // numSampleSizeEntries
          null,                           // sampleSizeArray
          &sample_buffer,                 // sampleBufferOut
      );

      if (status != vtb.noErr) {
          return error.CreateSampleBufferFailed;
      }

      return sample_buffer.?;
  }
